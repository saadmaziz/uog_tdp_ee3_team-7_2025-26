#include "mbed.h"

// Hardware Pin Assignments
I2C i2c(PTC2, PTC1);         // SDA, SCL
DigitalOut en(PTE4);         // Sensor Enable pin
PwmOut led_r(LED_RED);       // RGB Red (Active Low)
PwmOut led_g(LED_GREEN);     // RGB Green (Active Low)
PwmOut pwm_out(PTE5);       // Variable frequency PWM

const int ADDR = 0x41 << 1;  // TMF8806 I2C Address [cite: 764]

void write(uint8_t reg, uint8_t val) {
    char data[] = {reg, val};
    i2c.write(ADDR, data, 2);
}

int main() {
    // 1. HARDWARE STARTUP [cite: 435, 463]
    en = 0; ThisThread::sleep_for(10ms);
    en = 1; ThisThread::sleep_for(2ms); // Wait for I2C to be ready

    // 2. BOOTLOADER -> APP TRANSITION [cite: 450, 884]
    write(0xE0, 0x01); // Wake up oscillator (PON=1) [cite: 892]
    ThisThread::sleep_for(5ms);
    write(0x02, 0xC0); // Request Measurement App 
    ThisThread::sleep_for(5ms);

    // 3. START MEASUREMENT [cite: 478, 942]
    // Starts 2.5m mode continuous ranging [cite: 569, 996]
    char start[] = {0x06, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x06, 0x1E, 0x84, 0x02};
    i2c.write(ADDR, start, 11);

    while (true) {
        char res[33];
        char reg = 0x1D; // Read result registers starting at 0x1D [cite: 482, 1122]
        i2c.write(ADDR, &reg, 1, true);
        i2c.read(ADDR, res, 33);

        // Extract distance (mm) from registers 0x22 and 0x23 
        uint16_t dist = (res[6] << 8) | res[5];

        if (dist > 0) {
            // LED: Green if far (>1m), Red if close (<1m)
            led_r = (dist < 1000) ? 0.0f : 1.0f; 
            led_g = (dist >= 1000) ? 0.0f : 1.0f;

            // PWM: Vary period based on distance (closer = faster freq)
            pwm_out.period_ms(5 + (dist / 100)); // Period between 5ms and 55ms
            pwm_out = 0.5f; 
            
            printf("Distance: %u mm\n", dist);
        }

        write(0xE1, 0x01); // Clear interrupt flag [cite: 898]
        ThisThread::sleep_for(300ms); // 300ms = ~200 times per minute
    }
}
