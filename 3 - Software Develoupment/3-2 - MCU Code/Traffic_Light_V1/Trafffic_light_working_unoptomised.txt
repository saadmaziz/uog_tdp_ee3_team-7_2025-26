#include "mbed.h"

// I2C Address for VEML3328 (7-bit address 0x10 -> 8-bit write 0x20, read 0x21)
#define VEML3328_ADDR 0x20 

// Register Map
#define REG_CONFIG  0x00
#define REG_R_DATA  0x05
#define REG_G_DATA  0x06
#define REG_B_DATA  0x07

// Initialize I2C (Pins for FRDM-KL25Z: PTC, PTB2)
I2C i2c(PTC2, PTC1); 

// Initialize Onboard RGB LED (PWM)
// Note: KL25Z LEDs are active LOW (0.0 = bright, 1.0 = off)
PwmOut led_R(LED_RED);
PwmOut led_G(LED_GREEN);
PwmOut led_B(LED_BLUE);

int roundingfactor = 30;

// Helper function to read a 16-bit register from the sensor
uint16_t read_reg(uint8_t reg) {
    char cmd[1] = {reg};
    char data[2] = {0, 0};
    i2c.write(VEML3328_ADDR, cmd, 1, true);
    i2c.read(VEML3328_ADDR, data, 2);
    return (data[1] << 8) | data[0]; // Little-endian
}

int main() {
    // 1. Initialize Sensor: Power on, Gain x1, IT 100ms
    // Config: 0x0000 (SD=0 powers on)
    char config_cmd[3] = {REG_CONFIG, 0x00, 0x00};
    i2c.write(VEML3328_ADDR, config_cmd, 3);
    
    // Set PWM Period
    led_R.period(0.001f); 
    
    while (true) {
        // 2. Read Raw Color Data
        uint16_t r_raw = read_reg(REG_R_DATA);
        uint16_t g_raw = read_reg(REG_G_DATA);
        uint16_t b_raw = read_reg(REG_B_DATA);

        

        // 3. Normalize values (0.0 to 1.0)
        // The VEML3328 is 16-bit, but actual max depends on gain/integration.
        // We'll use 0xFFFF as a safety ceiling, or auto-scale based on the highest reading.
        float r_norm = round(((float)r_raw / 65535.0f) * roundingfactor)/roundingfactor;
        float g_norm = round(((float)g_raw / 65535.0f) * roundingfactor)/roundingfactor;
        float b_norm = round(((float)b_raw / 65535.0f) * roundingfactor)/roundingfactor;

        // 4. Output to RGB LED
        // KL25Z LED is common anode: Pulse Width = 1.0 - brightness
        led_R = 1.0f - r_norm;
        led_G = 1.0f - g_norm;
        led_B = 1.0f - b_norm;

        ThisThread::sleep_for(200ms);
    }
}
